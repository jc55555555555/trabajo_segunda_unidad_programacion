# Limpiar nombres duplicados
data <- data[, !duplicated(names(data))]

data$tratamiento <- as.factor(data$tratamiento)  # cada repetición = genotipo

# Verificar que solo haya 20 niveles
levels(data$tratamiento)  # debe dar 20
